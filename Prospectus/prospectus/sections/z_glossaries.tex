%% List of glossaries
\newglossaryentry{alpha}{
name=Reduced Deflection Angle ($\boldsymbol{\alpha}$ ),
description={$$
\boldsymbol{\alpha} (\boldsymbol{\theta}  ) \equiv \frac{4GM}{c^2}\  \frac{D_{ds}}{D_s D_d}\  \frac{\boldsymbol{\theta} }{\abs{\boldsymbol{\theta}}^2}
$$}}

\newglossaryentry{beta}{
name=True Position Of The Source ($\boldsymbol{\beta}$ ),
description={$$
\boldsymbol{\beta} = \boldsymbol{\theta} - \boldsymbol{\alpha}
$$}}

\newglossaryentry{theta}{
name=Observed Position Of The Source ($\boldsymbol{\theta}$ ),
description={$$
\boldsymbol{\beta} = \boldsymbol{\theta} - \boldsymbol{\alpha}
$$}}

\newglossaryentry{xi}{
name=Impact Parameter ($\boldsymbol{\xi}$ ),
description={$$
\boldsymbol{\theta} = \frac{\boldsymbol{\xi} }{D_d}
$$}}

\newglossaryentry{I}{
name=Observed Surface Brightness ($\boldsymbol{I}$ ),
description={$$
\boldsymbol{I}(\boldsymbol{\theta}) = \boldsymbol{I}^s[\beta_0 + \mathscr{A}(\boldsymbol{\theta}) (\theta - \theta_0 ) ]
$$}}

\newglossaryentry{surface_mass_density}{
name=Surface Mass Density ($\Sigma$),
description={$$
	\Sigma(\boldsymbol{\xi}) \equiv \int dr_3 \ \rho(\xi_1, \xi_2, r_3)
$$}}

\newglossaryentry{dist_matrix}{
name=Distortion Matrix ($\mathscr{A}$),
description={$$
  \mathscr{A}(\boldsymbol{\theta} ) \equiv \frac{\partial \boldsymbol{\beta} }{ \partial \boldsymbol{\theta} }
$$}}

\newglossaryentry{shear}{
name=Shear ($\gamma$),
description={$$
\gamma \equiv \gamma_1 + i \gamma_2
			 = \lvert\gamma\rvert e^{2i \phi}
$$}}

\newglossaryentry{reduced_shear}{
name=Reduced Shear (g),
description={$$
g(\boldsymbol{\theta} ) \equiv \frac{\gamma(\boldsymbol{\theta} )}{1 - \kappa(\boldsymbol{\theta} )}
$$}}

\newglossaryentry{magnification}{
name=Magnification ($\mu$),
description={$$
\mu(\boldsymbol{\theta}) \equiv \frac{1}{det \lvert \mathscr{A} \rvert }
$$}}
